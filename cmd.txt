git reset  --hard HEAD^                    |  版本回退
git reset  --hard HEAD~100              |  版本回退100次
git status                                           | 查看是否有未提交的修改
git add "文件名"                                 | 把修改的文件添加到缓存区
git log                                                | 查看修改历史
git commit -m "文件名"                      | 提交缓存区文件到版本库
